PremierMatches = filter(MatchesData, MatchesData$league_id == "148")

BetsData1 = filter(BetsData, BetsData$odd_bookmakers == "1xBet" & BetsData$match_id %in% PremierMatches$match_id )
BetsData1Home = filter(BetsData1, BetsData1$variable == "odd_1")
BetsData1Away = filter(BetsData1, BetsData1$variable == "odd_2")
BetsData1Draw = filter(BetsData1, BetsData1$variable == "odd_x")

###Part 1 Calculating the probabilities
PHomeWin1 = 1/BetsData1Home$value
PAwayWin1 = 1/BetsData1Away$value
PDraw1 = 1/BetsData1Draw$value

###Part 2 Normalization of probabilities
PHomeWin1Normal = PHomeWin1 / (PHomeWin1 + PAwayWin1 + PDraw1)
PAwayWin1Normal = PAwayWin1 / (PHomeWin1 + PAwayWin1 + PDraw1)
PDraw1Normal = PDraw1 / (PHomeWin1 + PAwayWin1 + PDraw1)

###Part 3
DrawMatches = filter(MatchesData, MatchesData$match_hometeam_score - MatchesData$match_awayteam_score == 0 & MatchesData$league_id == "148")
plot(PHomeWin1-PAwayWin1, PDraw1,xlab= "P(Home)-P(Away)", ylab = "P(Draw)",col="red",main="1xBet")
plot(PHomeWin1Normal-PAwayWin1Normal,PDraw1Normal,xlab= "P(Home)-P(Away) Normalized", ylab = "P(Draw) Normalized",col="blue",main="1xBet")
TotalCountBin1=list()
CountBin1 = list()
MyProbability1 = list()
Bias1 = list()
for(i in 1:10){
  TotalCountBin1[i] = sum((PHomeWin1-PAwayWin1)>=(-1.2+0.2*i) & (PHomeWin1-PAwayWin1)<(-1.2+0.2*(i+1)))
}

BetsData1ActualDraw = filter(BetsData1, BetsData1$match_id %in% DrawMatches$match_id)

BetsData1ActualDrawHome = filter(BetsData1ActualDraw, BetsData1ActualDraw$variable == "odd_1")

BetsData1ActualDrawAway = filter(BetsData1ActualDraw, BetsData1ActualDraw$variable == "odd_2")

PHomeWinActual1 = 1/BetsData1ActualDrawHome$value
PAwayWinActual1 = 1/BetsData1ActualDrawAway$value

for(i in 1:10){
  CountBin1[i] = sum((PHomeWinActual1-PAwayWinActual1)>=(-1.2+0.2*i) & (PHomeWinActual1-PAwayWinActual1)<(-1.2+0.2*(i+1)))
}

for(i in 1:10){
  MyProbability1[i] = CountBin1[[i]]/TotalCountBin1[[i]]
}

for(i in 1:10){
  Bias1[i] = sum((PHomeWin1-PAwayWin1)>(-1.2+0.2*i) & (PHomeWin1-PAwayWin1)<=(-1.2+0.2*(i+1)) & MyProbability1[i]>PDraw1 )
}
BiasArray1 = unlist(Bias1)
plot(x=1:10, y=BiasArray1,xlab="Index of the bin",ylab="Number of Biased Odds",main="1xBet Bias Graph",xaxt = "n",yaxt="n")
axis(1, at = seq(1, 10, by = 1), las=2)
axis(2, at=seq(0,max(BiasArray1)+1,by=5), las=2)
